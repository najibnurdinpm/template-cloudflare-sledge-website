export const ratingComponentCode = `"use client";

import { Rating } from "@sledge-app/react-product-review";

type iRatingComponent = {
  size?: "md" | "sm" | "xs";
  productId?: any;
};
export default function RatingComponent({
  size = "sm",
  productId,
}: iRatingComponent) {
  return <Rating size={size} params={{ productId }} />;
}`;

    // ...

  )
}`;

export const widgetComponentCode = `"use client";

import { Widget, WidgetHeader } from "@sledge-app/react-product-review";

type IReviewList = {
  productId: string;
};
export default function ReviewWidget({ productId }: IReviewList) {
  return (
    <Widget.Root params={{ productId }} >
      <Widget.Header>
        <WidgetHeader.Summary />
        <WidgetHeader.AddTrigger />
        <WidgetHeader.Sort />
      </Widget.Header>
      <Widget.List />
    </Widget.Root>
  );
}`;
