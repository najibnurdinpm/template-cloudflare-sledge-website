export const iconComponentCode = `"use client";

import { CustomComponents } from "@sledge-app/core";
import { SearchIconWidget } from "@sledge-app/react-instant-search";
import Link from "next/link";

function viewMoreResultButton({ keyword, setShowPopupComponent }: any){
    return (
      <Link
        // fill href with search result page url
        href={\`/search-result?q=\${keyword}\`}
        onClick={() => setShowPopupComponent && setShowPopupComponent(false)}
      >View More Result</Link>
    );
}

export default function SearchHeaderMenu() {
  return (
    <SearchIconWidget size="sm">
      <CustomComponents 
        searchViewMoreResult={ viewMoreResultButton }
      />
    </SearchIconWidget>
  );
}`;
